1) 	SELECT InvoiceID,InvoiceNumber,InvoiceTotal,PaymentTotal,CreditTotal 
	FROM Invoices
	WHERE PaymentTotal > 0 OR CreditTotal > 0


2)	SELECT VendorID, VendorName, VendorCity, VendorState 
	FROM Vendors
	WHERE VendorState IN('AZ','NJ','NV','OH')
	--WHERE VendorState = 'AZ' OR  VendorState = 'NJ' OR VendorState ='NV' OR VendorState ='OH'
	ORDER BY VendorState, VendorCity


3)	SELECT InvoiceID, VendorID, InvoiceDate, InvoiceTotal 
	FROM Invoices
	WHERE VendorID = 123 
	AND InvoiceDate < '2016-01-01'
	ORDER BY InvoiceDate ASC


4)	SELECT * 
	FROM GLAccounts
	WHERE AccountDescription LIKE '%book%' 
	AND NOT AccountDescription LIKE '%books%'


5)	SELECT VendorID, VendorName, VendorPhone, VendorContactFName +', ' + VendorContactLName AS VendorContact
	FROM Vendors
	WHERE VendorPhone IS NULL


6)	SELECT DISTINCT(InvoiceLineItemDescription) 
	FROM InvoiceLineItems
	WHERE InvoiceLineItemDescription LIKE 'C%'
